Basic Text Generation (simplest)```kotlinimport com.google.ai.client.generativeai.GenerativeModelsuspend fun generateText(prompt: String): String {    val model = GenerativeModel(        modelName = "gemini-1.5-flash",        apiKey = BuildConfig.GEMINI_API_KEY    )    return model.generateContent(prompt).text ?: "No response"}```### 2. Multi-turn Chat```kotlinimport com.google.ai.client.generativeai.GenerativeModelimport com.google.ai.client.generativeai.Chatsuspend fun runChat() {    val model = GenerativeModel(        modelName = "gemini-1.5-flash",        apiKey = BuildConfig.GEMINI_API_KEY    )    val chat: Chat = model.startChat()    val reply1 = chat.sendMessage("Hello! What's the capital of France?")    println("Bot: ${reply1.text}")    val reply2 = chat.sendMessage("What's the population of that city?")    println("Bot: ${reply2.text}")}```### 3. Streaming Response```kotlinimport com.google.ai.client.generativeai.GenerativeModelimport kotlinx.coroutines.flow.collectsuspend fun streamResponse(prompt: String) {    val model = GenerativeModel(        modelName = "gemini-1.5-flash",        apiKey = BuildConfig.GEMINI_API_KEY    )    model.generateContentStream(prompt).collect { chunk ->        print(chunk.text)   // tokens arrive incrementally    }}```### 4. Image + Text (Multimodal / Vision)```kotlinimport android.graphics.Bitmapimport com.google.ai.client.generativeai.GenerativeModelimport com.google.ai.client.generativeai.type.contentsuspend fun analyzeImage(bitmap: Bitmap, question: String): String {    val model = GenerativeModel(        modelName = "gemini-1.5-flash",   // vision-capable model        apiKey = BuildConfig.GEMINI_API_KEY    )    val inputContent = content {        image(bitmap)        text(question)    }    return model.generateContent(inputContent).text ?: "No response"}```### 5. Generation Config + Safety Settings```kotlinimport com.google.ai.client.generativeai.GenerativeModelimport com.google.ai.client.generativeai.type.*suspend fun generateWithConfig(prompt: String): String {    val config = generationConfig {        temperature = 0.7f        topK = 40        topP = 0.95f        maxOutputTokens = 1024    }    val safetySettings = listOf(        SafetySetting(HarmCategory.HARASSMENT, BlockThreshold.MEDIUM_AND_ABOVE),        SafetySetting(HarmCategory.HATE_SPEECH, BlockThreshold.MEDIUM_AND_ABOVE)    )    val model = GenerativeModel(        modelName = "gemini-1.5-flash",        apiKey = BuildConfig.GEMINI_API_KEY,        generationConfig = config,        safetySettings = safetySettings    )    return model.generateContent(prompt).text ?: "No response"}```### 6. Android ViewModel (production pattern)```kotlinimport androidx.lifecycle.ViewModelimport androidx.lifecycle.viewModelScopeimport com.google.ai.client.generativeai.GenerativeModelimport kotlinx.coroutines.flow.MutableStateFlowimport kotlinx.coroutines.flow.StateFlowimport kotlinx.coroutines.launchclass GeminiViewModel : ViewModel() {    private val model = GenerativeModel(        modelName = "gemini-1.5-flash",        apiKey = BuildConfig.GEMINI_API_KEY    )    private val _uiState = MutableStateFlow<UiState>(UiState.Idle)    val uiState: StateFlow<UiState> = _uiState    fun ask(prompt: String) {        _uiState.value = UiState.Loading        viewModelScope.launch {            _uiState.value = try {                val text = model.generateContent(prompt).text ?: "No response"                UiState.Success(text)            } catch (e: Exception) {                UiState.Error(e.message ?: "Unknown error")            }        }    }    sealed class UiState {        object Idle : UiState()        object Loading : UiState()        data class Success(val text: String) : UiState()        data class Error(val message: String) : UiState()    }}```---## Model Selection Guide| Model | Best for ||---|---|| `gemini-1.5-flash` | Speed, cost efficiency, everyday tasks || `gemini-1.5-pro` | Complex reasoning, long context || `gemini-1.0-pro` | Legacy / stable baseline |Default to **`gemini-1.5-flash`** unless the user specifies otherwise.---## Key Rules to Always Follow1. All `generateContent` / `sendMessage` calls are **`suspend` functions** — always call inside a coroutine (`viewModelScope.launch`, `lifecycleScope.launch`, or `runBlocking` for tests).2. Wrap calls in **`try/catch`** — handle `GoogleGenerativeAIException` or generic `Exception`.3. Never hardcode API keys — always use `BuildConfig` or environment variables.4. For streaming, collect the `Flow` with `.collect { }` inside a coroutine.5. For multimodal, use `gemini-1.5-flash` or higher (vision-capable).---## Quick Reference: Common Errors| Error | Fix ||---|---|| `NetworkOnMainThreadException` | Move call to coroutine / background thread || `API_KEY_INVALID` | Check key in AI Studio; confirm `BuildConfig` setup || `PERMISSION_DENIED` | Enable Generative Language API in Google Cloud Console || `Response was null` | Use `?.text ?: "fallback"` — response can be null if blocked by safety filters |---For full docs, see: https://ai.google.dev/gemini-api/docs/quickstart?lang=android
